 What is statics?
statics are functions you call on the Model itself (not on individual documents).

Opposite of methods (which you call on documents).


| Type      | Called on  | Example                   |
| --------- | ---------- | ------------------------- |
| `methods` | a document | `user.sayHello()`         |
| `statics` | the Model  | `User.findByEmail(email)` |


userSchema.statics.findByEmail = function(email) {
  return this.findOne({ email });
};

Here, findByEmail is a custom function added to the User model.

const User = mongoose.model('User', userSchema);

User.findByEmail("abc@gmail.com").then(user => {
  console.log(user); // user document with that email
});



const userSchema = new mongoose.Schema({
  name: String,
  email: String,
  address: {
    city: String,
    country: String
  }
});

userSchema.statics.findByCountry = function(countryName) {
  return this.find({ 'address.country': countryName });
};

User.findByCountry("India").then(users => {
  console.log(users);
});


const User = mongoose.model('User', userSchema);

User.findByCity("Kochi").then(users => {
  console.log("Users in Kochi:", users);
});
