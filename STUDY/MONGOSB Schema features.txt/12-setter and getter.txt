What are Setters & Getters?
Setters → Modify value before saving to DB

Getters → Modify value after reading from DB

userSchema.path('name').set(function(v) {
  return v.trim();  // remove extra spaces
});

userSchema.path('email').set(function (value) {
  return value.toLowerCase();
});

productSchema.path('price').set(function (value) {
  return Math.round(value);
});

userSchema.path('phone').set(function (value) {
  return value.replace(/[\s-]/g, '');
});

orderSchema.path('createdAt').get(function (value) {
  return value.toDateString(); // "Thu Jul 04 2025"
});

userSchema.path('name').get(function (value) {
  return value.charAt(0).toUpperCase() + value.slice(1);
});


productSchema.path('price').get(function (value) {
  return `₹${value}`;
});

To enable getters on JSON output:
schema.set('toJSON', { getters: true });
schema.set('toObject', { getters: true }); // (optional)
